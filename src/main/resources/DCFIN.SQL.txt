CREATE SCHEMA `dcfinancial` DEFAULT CHARACTER SET utf8 ;

CREATE TABLE `dcfinancial`.`dcf_idp_user` (
  `DCF_IDP_USER_ID` VARCHAR(50) NOT NULL,
  `DCF_IDP_USER_UNAME` VARCHAR(10) NULL,
  `DCF_IDP_USER_EMAIL` VARCHAR(25) NULL,
  `DCF_IDP_USER_FNAME` VARCHAR(10) NULL,
  `DCF_IDP_USER_MNAME` VARCHAR(10) NULL,
  `DCF_IDP_USER_LNAME` VARCHAR(10) NULL,
  `DCF_IDP_USER_ADDRESS` VARCHAR(45) NULL,
  `DCF_IDP_USER_CREATE_TIME` DATETIME NULL,
  `DCF_IDP_USER_CREATE_BY` VARCHAR(10) NULL,
  `DCF_IDP_USER_UPDATE_TIME` DATETIME NULL,
  `DCF_IDP_USER_UPDATE_BY` VARCHAR(10) NULL,
  `DCF_IDP_USER_DELETE_TIME` DATETIME NULL,
  `DCF_IDP_USER_DELETE_BY` VARCHAR(10) NULL,
  `DCF_IDP_USER_IS_ACTIVE` TINYINT NULL,
  PRIMARY KEY (`DCF_IDP_USER_ID`),
  UNIQUE INDEX `DCF_IDP_USER_ID_UNIQUE` USING BTREE (`DCF_IDP_USER_ID`, `DCF_IDP_USER_UNAME`, `DCF_IDP_USER_EMAIL`, `DCF_IDP_USER_CREATE_TIME`, `DCF_IDP_USER_FNAME`, `DCF_IDP_USER_MNAME`, `DCF_IDP_USER_LNAME`, `DCF_IDP_USER_ADDRESS`, `DCF_IDP_USER_IS_ACTIVE`) INVISIBLE)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
COMMENT = 'User Information ';
